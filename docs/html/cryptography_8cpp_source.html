252void generate(const uint8_t key[32], uint8_t round_keys[10][block_len]) {
253 // получаем 32 константы: 4 итерации по 8 констант каждая
254 uint8_t round_constants[32][block_len] = {};
255 generate_round_constants(round_constants);
256
257 // первые 2 ключа - половинки исходного
258 memcpy(round_keys[0], key + (uint8_t) block_len, block_len);
259 memcpy(round_keys[1], key, block_len);
260
261 // сеть Фейстеля
262 for (uint8_t i = 1; i < 5; ++i) {
263 uint8_t left[block_len] = {};
264 uint8_t right[block_len] = {};
265 uint8_t new_left[block_len] = {};
266
267 memcpy(left, round_keys[2 * (i - 1)], block_len);
268 memcpy(right, round_keys[2 * (i - 1) + 1], block_len);
269
270 for (uint8_t j = 0; j < 8; ++j) {
271 for (uint8_t k = 0; k < block_len; ++k) {
272 new_left[k] = S[(left[k] ^ round_constants[(i - 1) * 8 + j][k])];
273 }
274 L_transform(new_left);
275 for (uint8_t k = 0; k < block_len; ++k) {
276 new_left[k] ^= right[k];
277 }
278 memcpy(right, left, block_len);
279 memcpy(left, new_left, block_len);
280 }
281 memcpy(round_keys[2 * i], left, block_len);
282 memcpy(round_keys[2 * i + 1], right, block_len);
283 }
284}

376void encode_block(uint8_t block[block_len], const uint8_t round_keys[10][block_len]) {

431std::vector<uint8_t> cbcEncrypt(const std::vector<uint8_t> &plaintext, const uint8_t round_keys[10][block_len]) {
432 // Инициализационный вектор (IV)
433 uint8_t iv[block_len] = {
434 0xa5, 0x2D, 0x32, 0x8F, 0x0E, 0x30, 0x38, 0xC0,
435 0x54, 0xE6, 0x9E, 0x39, 0x55, 0x7E, 0x52, 0x91
436 };
437
438 // Расчет количества блоков
439 size_t num_blocks = (plaintext.size() + block_len - 1) / block_len;
440 std::vector<uint8_t> ciphertext(num_blocks * block_len);
441
442 // Предыдущий блок, используется для XOR с текущим блоком (начинаем с IV)
443 uint8_t prev_block[block_len];
444 std::memcpy(prev_block, iv, block_len);
445
446 for (size_t i = 0; i < num_blocks; ++i) {
447 uint8_t block[block_len] = {};
448 size_t block_start = i * block_len;
449
450 // Копирование данных открытого текста в блок с учетом возможного неполного блока
451 for (size_t j = 0; j < block_len && block_start + j < plaintext.size(); ++j) {
452 block[j] = plaintext[block_start + j];
453 }
454
455 // XOR с предыдущим блоком или IV
456 xorBlocks(block, prev_block, block_len);
457
458 // Шифрование блока
459 encode_block(block, round_keys);
460
461 // Копирование зашифрованного блока в результат
462 std::memcpy(&ciphertext[block_start], block, block_len);
463
464 // Обновление предыдущего блока для следующей итерации
465 std::memcpy(prev_block, block, block_len);
466 }
467
468 return ciphertext;
469}

477std::vector<uint8_t> cbcDecrypt(const std::vector<uint8_t> &ciphertext, const uint8_t round_keys[10][block_len]) {
478 uint8_t iv[block_len] = {0xa5, 0x2D, 0x32, 0x8F, 0x0E, 0x30, 0x38, 0xC0, 0x54, 0xE6, 0x9E, 0x39, 0x55, 0x7E, 0x52,
479 0x91};
480 size_t num_blocks = ciphertext.size() / block_len;
481 std::vector<uint8_t> plaintext(num_blocks * block_len);
482 uint8_t prev_block[block_len];
483 std::memcpy(prev_block, iv, block_len);
484 for (size_t i = 0; i < num_blocks; ++i) {
485 uint8_t block[block_len] = {};
486 size_t block_start = i * block_len;
487 std::memcpy(block, &ciphertext[block_start], block_len);
488 uint8_t decrypted_block[block_len];
489 std::memcpy(decrypted_block, block, block_len);
490
491 decode_block(decrypted_block, round_keys);
492
493 // XOR с предыдущим блоком или IV
494 xorBlocks(decrypted_block, prev_block, block_len);
495
496 std::memcpy(&plaintext[block_start], decrypted_block, block_len);
497 std::memcpy(prev_block, block, block_len);
498 }
499
500 // Удаление дополнения
501 while (!plaintext.empty() && plaintext.back() == 0) {
502 plaintext.pop_back();
503 }
504
505 return plaintext;
506}
